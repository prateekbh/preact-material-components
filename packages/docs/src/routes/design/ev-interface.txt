interface IEventInfo {
    component: MaterialComponent;  // The instance of the component that emits the event
}

interface IProps {
    /**
     * A event handler
     * @param event: The event emitted by the browser/material
     * @param additional: Additional information for the handler
     * @returns: Nothing
     */
    onSomeEvent: (event: Event, additional: IEventInfo) => void;
}
